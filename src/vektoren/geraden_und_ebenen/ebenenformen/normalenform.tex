
\subsubsection{Normalenform}

$
\begin{pmatrix}
    (\vec{p} - \vec{x}) \bullet \vec{n}
\end{pmatrix}
$ 
\\

$
\begin{pmatrix}
    \begin{Bmatrix}
        p_1 \\ p_2 \\ p_3
    \end{Bmatrix}

    \begin{Bmatrix}
        x_1 \\ x_2 \\ x_3
    \end{Bmatrix}
\end{pmatrix}
\bullet 
\begin{Bmatrix}
    n_1 \\ n_2 \\ n_3
\end{Bmatrix}
$
